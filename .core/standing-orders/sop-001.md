## 2. WEEKLY MAINTENANCE SCHEDULE

### **SUNDAY 02:00 UTC - PRIMARY MAINTENANCE**

```bash
#!/bin/bash
# Weekly System Maintenance - SUNDAY WINDOW
# Execute during low-activity period

echo "=== WEEKLY SYSTEM MAINTENANCE $(date) ==="

# 1. SWAP & MEMORY CHECK
free -h
echo "Swap Usage: $(cat /proc/swaps | tail -1 | awk '{printf "%.1f%%\n", $3/$2*100}')"

# 2. DISK USAGE ASSESSMENT  
df -h
echo "Critical partitions >80%:"
df -h | awk '$5+0 > 80 {print $0}'

# 3. CACHE CLEANUP (Priority 1)
echo "--- CACHE CLEANUP ---"
du -sh ~/.cache/* 2>/dev/null | sort -hr | head -5
rm -rf ~/.cache/vscode-cpptools/ipch/* 2>/dev/null
rm -rf ~/.cache/mozilla/firefox/*/cache2/* 2>/dev/null
rm -rf ~/.cache/pip/wheels/* 2>/dev/null
echo "Cache cleaned"

# 4. LOG ROTATION ENFORCEMENT
echo "--- LOG MANAGEMENT ---"
journalctl --disk-usage
journalctl --vacuum-time=7d --vacuum-size=500M
sudo logrotate -f /etc/logrotate.conf

# 5. DOCKER CLEANUP
echo "--- DOCKER MAINTENANCE ---"
docker system prune -f --volumes 2>/dev/null || echo "Docker not accessible"

# 6. TEMP FILE PURGE
echo "--- TEMP FILE CLEANUP ---"
find /tmp -type f -mtime +2 -delete 2>/dev/null
find ~/.local/share/Trash -type f -mtime +7 -delete 2>/dev/null

# 7. LARGE FILE DETECTION
echo "--- LARGE FILE AUDIT ---"
find /home/adam -type f -size +500M -ls 2>/dev/null | head -5

echo "=== MAINTENANCE COMPLETE $(date) ==="
```

### **WEDNESDAY 14:00 UTC - MID-WEEK CHECK**

```bash
#!/bin/bash
# Mid-week Health Check - WEDNESDAY WINDOW

echo "=== MID-WEEK HEALTH CHECK $(date) ==="

# Quick status check
SWAP_PCT=$(cat /proc/swaps | tail -1 | awk '{printf "%.0f", $3/$2*100}')
ROOT_PCT=$(df / | tail -1 | awk '{print $5}' | tr -d '%')

echo "Swap: ${SWAP_PCT}% | Root Disk: ${ROOT_PCT}%"

if [ "$SWAP_PCT" -gt 90 ]; then
    echo "🚨 CRITICAL: Swap >90% - Emergency cleanup needed"
    # Emergency cache flush
    sync && echo 3 > /proc/sys/vm/drop_caches 2>/dev/null || echo "Cache drop failed (requires sudo)"
fi

if [ "$ROOT_PCT" -gt 85 ]; then
    echo "⚠️ WARNING: Root disk >85%"
    du -sh /var/log /tmp ~/.cache | sort -hr
fi

echo "=== CHECK COMPLETE $(date) ==="
```
